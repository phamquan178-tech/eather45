2. Replace components/BrandLogo.tsx with an image-based component:

import Image from "next/image";

type BrandLogoProps = {
  variant?: "dark" | "light" | "white";
  size?: "small" | "medium" | "large";
  className?: string;
};

export default function BrandLogo({
  variant = "dark",
  size = "medium",
  className = "",
}: BrandLogoProps) {
  const src =
    variant === "light"
      ? "/images/eather45/logo-light.png"
      : variant === "white"
      ? "/images/eather45/logo-white.png"
      : "/images/eather45/logo-dark.png";

  const sizeMap = {
    small: { width: 150, height: 58 },
    medium: { width: 280, height: 108 },
    large: { width: 440, height: 170 },
  } as const;

  return (
    <Image
      src={src}
      alt="Eather45 logo"
      width={sizeMap[size].width}
      height={sizeMap[size].height}
      priority
      className={`h-auto w-auto object-contain ${className}`}
    />
  );
}
