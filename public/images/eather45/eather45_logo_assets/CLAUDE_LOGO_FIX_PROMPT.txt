import Image from "next/image";

type BrandLogoProps = {
  variant?: "light" | "dark";
  size?: "small" | "medium" | "large";
  iconOnly?: boolean;
  className?: string;
};

const logoSizes = {
  small: { width: 170, height: 55 },
  medium: { width: 300, height: 98 },
  large: { width: 460, height: 150 },
};

const iconSizes = {
  small: { width: 72, height: 29 },
  medium: { width: 120, height: 48 },
  large: { width: 180, height: 72 },
};

export default function BrandLogo({
  variant = "dark",
  size = "medium",
  iconOnly = false,
  className = "",
}: BrandLogoProps) {
  const src = iconOnly
    ? variant === "light"
      ? "/images/eather45/icon-light.svg"
      : "/images/eather45/icon-dark.svg"
    : variant === "light"
      ? "/images/eather45/logo-light.svg"
      : "/images/eather45/logo-dark.svg";

  const dimensions = iconOnly ? iconSizes[size] : logoSizes[size];

  return (
    <Image
      src={src}
      alt="Eather45"
      width={dimensions.width}
      height={dimensions.height}
      priority
      className={`object-contain ${className}`}
    />
  );
}
